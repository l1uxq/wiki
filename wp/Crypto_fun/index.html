    Crypto-2EM¶

    看一下代码，secret是getrandbits(32)，经典的MT19937预测伪随机数问题。

    两次使用了getrandbits(32)，由于给了2**22组，只需要624组即可预测之前的随机数。找一下for循环之前的随机数即可。

    #!/usr/bin/env python2
from Crypto.Util.number import *
from random import Random

def inverse_right(res,shift,bits=32):
    tmp = res
    for i in range(bits//shift):
        tmp = res ^ tmp >> shift
    return tmp
# right shift with mask inverse
def inverse_right_values(res,shift,mask,bits=32):
    tmp = res
    for i in range(bits//shift):
        tmp = res ^ tmp>>shift & mask
    return tmp
# left shift inverse
def inverse_left(res,shift,bits=32):
    tmp = res
    for i in range(bits//shift):
        tmp = res ^ tmp << shift
    return tmp
# left shift with mask inverse
def inverse_left_values(res,shift,mask,bits=32):
    tmp = res
    for i in range(bits//shift):
        tmp = res ^ tmp << shift & mask
    return tmp


def backtrace(cur):
    high = 0x80000000
    low = 0x7fffffff
    mask = 0x9908b0df
    state = cur
    for i in range(3,-1,-1):
        tmp = state[i+624]^state[i+397]
        # recover Y,tmp = Y
        if tmp & high == high:
            tmp ^= mask
            tmp <<= 1
            tmp |= 1
        else:
            tmp <<=1
        # recover highest bit
        res = tmp&high
        # recover other 31 bits,when i =0,it just use the method again it so beautiful!!!!
        tmp = state[i-1+624]^state[i+396]
        # recover Y,tmp = Y
        if tmp & high == high:
            tmp ^= mask
            tmp <<= 1
            tmp |= 1
        else:
            tmp <<=1
        res |= (tmp)&low
        state[i] = res
    return state

def recover_state(out):
    state = []
    for i in out:
        i = inverse_right(i,18)
        i = inverse_left_values(i,15,0xefc60000)
        i = inverse_left_values(i,7,0x9d2c5680)
        i = inverse_right(i,11)
        state.append(i)
    return state

def d_p(data,pbox):
    tmp = bin(data)[2:].rjust(32,'0')
    out = [x for x in range(32)]
    for i,j in zip(pbox,tmp):
        out[i] = j
    return int(''.join(out),2)

key = 1991722937

for i in a:
    tmp2 = i^key
    tmp1 = d_p(tmp2,pbox2)^key
    msg = d_p(tmp1,pbox1)^key
    print(long_to_bytes(int(bin(msg)[2:],2)).decode(),end='')
